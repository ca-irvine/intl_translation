import 'dart:ui';

import 'package:intl/intl.dart';

class Message {
  const Message._(this.locale);

  factory Message.of(Locale locale) {
    return Message._(locale);
  }

  final Locale locale;

  String get title {
    return Intl.message(
      "Title is here(de)",
      name: "title",
    );
  }

  String content(dynamic name) {
    return Intl.message(
      "Hello, {name} (de)",
      name: "content",
      args: [name],
    );
  }

}
